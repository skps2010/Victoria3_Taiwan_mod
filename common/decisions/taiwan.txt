han_claim_china = {
	is_shown = {
		any_primary_culture = { has_discrimination_trait = han_chinese }
		NOT = {
			c:CHI ?= this
			has_variable = claim_china_var
		}
		is_revolutionary = no
	}

	possible = {
		exists = c:CHI
		NOT = {
			is_subject_of = c:CHI
		}
		country_rank >= c:CHI.country_rank
		c:CHI = { relations:root <= relations_threshold:poor }	
	}

	when_taken = {
		set_variable = claim_china_var
		trigger_event = {
			id = claim_china.1
			days = 0
			popup = yes
		}
	}

	ai_chance = {
		value = 100
	}
}

han_unclaim_china = {
	is_shown = {
		has_variable = claim_china_var
		NOT = {
			has_variable = unclaim_china_var
		}
	}

	possible = {
		exists = c:CHI
		c:CHI = { relations:root >= relations_threshold:cordial }	
	}

	when_taken = {
		set_variable = unclaim_china_var
		trigger_event = {
			id = claim_china.2
			days = 0
			popup = yes
		}
	}

	ai_chance = {
		value = 0
	}
}

			}

				owner = ROOT
			}
		}

			is_country_type = decentralized

		}
	}

	possible = {
	}

	when_taken = {
		s:STATE_FORMOSA = {
			add_claim = ROOT
		}
	}

	ai_chance = {
		value = 100
	}
}
